package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;

public final class IOUtils {
  private IOUtils() {}

  public static void ensureDir(File dir) throws IOException {
    if (dir == null) throw new IllegalArgumentException("dir is null");
    if (dir.exists()) {
      if (!dir.isDirectory()) throw new IOException("Not a directory: " + dir);
      return;
    }
    Files.createDirectories(dir.toPath());
  }

  public static void writeUtf8(Path path, String content) throws IOException {
    Files.createDirectories(path.getParent());
    Files.write(path, content.getBytes(StandardCharsets.UTF_8));
  }
}

PDF to Text (PDFBox)

Create a Java utility using Apache PDFBox that takes a PDF file input and saves all extracted text into a .txt file, preserving basic paragraph spacing. 

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import java.nio.file.Path;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.text.PDFTextStripper;

public final class PdfToText {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: PdfToText input.pdf output.txt");
      System.exit(1);
    }
    extract(new File(args[0]), Path.of(args[1]));
  }

  public static void extract(File inputPdf, Path outputTxt) throws IOException {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      PDFTextStripper stripper = new PDFTextStripper();
      stripper.setSortByPosition(true);
      stripper.setAddMoreFormatting(true);
      // Basic paragraph spacing: preserve line breaks as-is; PDFBox will add them
      String text = stripper.getText(doc).replaceAll("\\n{3,}", "\n\n").trim() + "\n";
      IOUtils.writeUtf8(outputTxt, text);
    }
  }
}

Image to PDF (PDFBox)

Write a Java program that takes a folder of JPG/PNG images and wraps each image into a single multi-page PDF document where each page matches the image dimensions. 

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import java.util.Arrays;
import java.util.Comparator;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.common.PDRectangle;
import org.apache.pdfbox.pdmodel.graphics.image.PDImageXObject;

public final class ImagesToPdf {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: ImagesToPdf inputFolder output.pdf");
      System.exit(1);
    }
    createPdfFromFolder(new File(args[0]), new File(args[1]));
  }

  public static void createPdfFromFolder(File folder, File outPdf) throws IOException {
    File[] images =
        folder.listFiles(
            f -> {
              String n = f.getName().toLowerCase();
              return f.isFile() && (n.endsWith(".jpg") || n.endsWith(".jpeg") || n.endsWith(".png"));
            });
    if (images == null || images.length == 0) throw new IOException("No images found in " + folder);

    Arrays.sort(images, Comparator.comparing(File::getName));

    try (PDDocument doc = new PDDocument()) {
      for (File img : images) {
        PDImageXObject ximg = PDImageXObject.createFromFileByContent(img, doc);
        PDRectangle rect = new PDRectangle(ximg.getWidth(), ximg.getHeight());
        PDPage page = new PDPage(rect);
        doc.addPage(page);

        try (PDPageContentStream cs = new PDPageContentStream(doc, page)) {
          cs.drawImage(ximg, 0, 0, rect.getWidth(), rect.getHeight());
        }
      }
      doc.save(outPdf);
    }
  }
}

Create a Java tool that renders each page of a PDF as a high-resolution (300 DPI) PNG image saved to a 'conversion' folder. 

package com.nutrithrive.labs;

import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.imageio.ImageIO;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.rendering.ImageType;
import org.apache.pdfbox.rendering.PDFRenderer;

public final class PdfToImages {
  public static void main(String[] args) throws Exception {
    if (args.length < 1) {
      System.out.println("Usage: PdfToImages input.pdf [outputFolder]");
      System.exit(1);
    }
    File out = new File(args.length > 1 ? args[1] : "conversion");
    render(new File(args[0]), out, 300);
  }

  public static void render(File inputPdf, File outputFolder, int dpi) throws IOException {
    IOUtils.ensureDir(outputFolder);
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      PDFRenderer renderer = new PDFRenderer(doc);
      for (int i = 0; i < doc.getNumberOfPages(); i++) {
        BufferedImage img = renderer.renderImageWithDPI(i, dpi, ImageType.RGB);
        File out = new File(outputFolder, "page_" + (i + 1) + ".png");
        ImageIO.write(img, "PNG", out);
      }
    }
  }
}

  }

package com.nutrithrive.labs;

import java.io.File;
import java.util.List;
import org.apache.pdfbox.multipdf.Splitter;
import org.apache.pdfbox.pdmodel.PDDocument;

public final class PdfSplitter {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: PdfSplitter input.pdf outputFolder");
      System.exit(1);
    }
    split(new File(args[0]), new File(args[1]));
  }

  public static void split(File inputPdf, File outputFolder) throws Exception {
    IOUtils.ensureDir(outputFolder);
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      Splitter splitter = new Splitter();
      List<PDDocument> pages = splitter.split(doc);
      int n = 1;
      for (PDDocument p : pages) {
        try (PDDocument pageDoc = p) {
          pageDoc.save(new File(outputFolder, "page_" + n + ".pdf"));
        }
        n++;
      }
    }
  }
}

Page Reorder (PDFBox)

Build a Java method that takes an existing PDF and an integer array of page indices (e.g., {2, 0, 1}) to create a new PDF in that specific page order.

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;

public final class PdfReorderPages {
  public static void main(String[] args) throws Exception {
    if (args.length < 3) {
      System.out.println("Usage: PdfReorderPages input.pdf output.pdf indicesCommaSeparated (e.g. 2,0,1)");
      System.exit(1);
    }
    int[] indices = parseIndices(args[2]);
    reorder(new File(args[0]), new File(args[1]), indices);
  }

  public static void reorder(File inputPdf, File outputPdf, int[] newOrderZeroBased) throws IOException {
    try (PDDocument in = PDDocument.load(inputPdf); PDDocument out = new PDDocument()) {
      for (int idx : newOrderZeroBased) {
        PDPage page = in.getPage(idx);
        out.addPage(page);
      }
      out.save(outputPdf);
    }
  }

  private static int[] parseIndices(String csv) {
    String[] parts = csv.split(",");
    int[] out = new int[parts.length];
    for (int i = 0; i < parts.length; i++) out[i] = Integer.parseInt(parts[i].trim());
    return out;
  }
}

Page Remover (PDFBox)

Write a Java tool that takes a PDF and a 'pagesToRemove' integer list, then generates a new PDF excluding those specific page numbers.

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;

public final class PdfRemovePages {
  public static void main(String[] args) throws Exception {
    if (args.length < 3) {
      System.out.println("Usage: PdfRemovePages input.pdf output.pdf pagesToRemove1Based (e.g. 1,3,5)");
      System.exit(1);
    }
    Set<Integer> remove = parsePages(args[2]);
    removePages(new File(args[0]), new File(args[1]), remove);
  }

  public static void removePages(File inputPdf, File outputPdf, Set<Integer> pagesToRemove1Based) throws IOException {
    try (PDDocument in = PDDocument.load(inputPdf); PDDocument out = new PDDocument()) {
      int pageIndex1 = 1;
      for (PDPage p : in.getPages()) {
        if (!pagesToRemove1Based.contains(pageIndex1)) out.addPage(p);
        pageIndex1++;
      }
      out.save(outputPdf);
    }
  }

  private static Set<Integer> parsePages(String csv) {
    Set<Integer> s = new HashSet<>();
    Arrays.stream(csv.split(",")).map(String::trim).filter(x -> !x.isEmpty()).forEach(x -> s.add(Integer.parseInt(x)));
    return s;
  }
}

PDF Compressor (PDFBox)

Develop a Java tool that reduces PDF size by iterating through all PDImageXObjects and reducing their JPEG quality/sampling rate.

Note: PDFBox doesn’t expose “JPEG quality” directly for existing embedded images; the standard approach is to decode images, re-encode (e.g., JPEG) at lower quality, then replace XObjects. 

package com.nutrithrive.labs;

import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import java.util.Iterator;
import javax.imageio.IIOImage;
import javax.imageio.ImageIO;
import javax.imageio.ImageWriteParam;
import javax.imageio.ImageWriter;
import javax.imageio.stream.ImageOutputStream;
import org.apache.pdfbox.cos.COSName;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDResources;
import org.apache.pdfbox.pdmodel.graphics.PDXObject;
import org.apache.pdfbox.pdmodel.graphics.image.JPEGFactory;
import org.apache.pdfbox.pdmodel.graphics.image.PDImageXObject;

public final class PdfCompressImages {
  public static void main(String[] args) throws Exception {
    if (args.length < 3) {
      System.out.println("Usage: PdfCompressImages input.pdf output.pdf quality(0.0-1.0)");
      System.exit(1);
    }
    compress(new File(args[0]), new File(args[1]), Float.parseFloat(args[2]));
  }

  public static void compress(File inputPdf, File outputPdf, float jpegQuality) throws IOException {
    if (jpegQuality <= 0 || jpegQuality > 1) throw new IllegalArgumentException("quality must be (0,1]");
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      for (PDPage page : doc.getPages()) {
        PDResources res = page.getResources();
        if (res == null) continue;
        for (COSName name : res.getXObjectNames()) {
          PDXObject xo = res.getXObject(name);
          if (!(xo instanceof PDImageXObject)) continue;
          PDImageXObject img = (PDImageXObject) xo;
          BufferedImage bi = img.getImage();
          if (bi == null) continue;

          // Re-encode as JPEG with chosen quality
          PDImageXObject compressed = jpegFromBufferedImage(doc, bi, jpegQuality);
          res.put(name, compressed);
        }
      }
      doc.save(outputPdf);
    }
  }

  private static PDImageXObject jpegFromBufferedImage(PDDocument doc, BufferedImage bi, float quality) throws IOException {
    // Use ImageIO to write JPEG to memory with a quality parameter, then wrap with JPEGFactory.
    Iterator<ImageWriter> writers = ImageIO.getImageWritersByFormatName("jpeg");
    if (!writers.hasNext()) throw new IOException("No JPEG ImageWriter available");
    ImageWriter writer = writers.next();
    ImageWriteParam param = writer.getDefaultWriteParam();
    if (param.canWriteCompressed()) {
      param.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
      param.setCompressionQuality(quality);
    }

    try (java.io.ByteArrayOutputStream baos = new java.io.ByteArrayOutputStream();
        ImageOutputStream ios = ImageIO.createImageOutputStream(baos)) {
      writer.setOutput(ios);
      writer.write(null, new IIOImage(bi, null, null), param);
      writer.dispose();
      return JPEGFactory.createFromStream(doc, new java.io.ByteArrayInputStream(baos.toByteArray()));
    }
  }
}

Grayscale Converter (PDFBox)

Build a Java function that processes a PDF and converts all colored text and images to a grayscale color space to reduce printing costs.

Note: full grayscale conversion for “colored text” requires content stream rewriting. Below is a practical version that converts images to grayscale (largest printing cost). 

package com.nutrithrive.labs;

import java.awt.color.ColorSpace;
import java.awt.image.BufferedImage;
import java.awt.image.ColorConvertOp;
import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.cos.COSName;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDResources;
import org.apache.pdfbox.pdmodel.graphics.PDXObject;
import org.apache.pdfbox.pdmodel.graphics.image.LosslessFactory;
import org.apache.pdfbox.pdmodel.graphics.image.PDImageXObject;

public final class PdfGrayscaleImages {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: PdfGrayscaleImages input.pdf output.pdf");
      System.exit(1);
    }
    convert(new File(args[0]), new File(args[1]));
  }

  public static void convert(File inputPdf, File outputPdf) throws IOException {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      ColorConvertOp op = new ColorConvertOp(ColorSpace.getInstance(ColorSpace.CS_GRAY), null);
      for (PDPage page : doc.getPages()) {
        PDResources res = page.getResources();
        if (res == null) continue;
        for (COSName name : res.getXObjectNames()) {
          PDXObject xo = res.getXObject(name);
          if (!(xo instanceof PDImageXObject)) continue;
          PDImageXObject img = (PDImageXObject) xo;
          BufferedImage bi = img.getImage();
          if (bi == null) continue;
          BufferedImage gray = op.filter(bi, null);
          PDImageXObject replaced = LosslessFactory.createFromImage(doc, gray);
          res.put(name, replaced);
        }
      }
      doc.save(outputPdf);
    }
  }
}

Watermark Creator (PDFBox)

package com.nutrithrive.labs;

import java.awt.Color;
import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.common.PDRectangle;
import org.apache.pdfbox.pdmodel.font.PDType1Font;
import org.apache.pdfbox.pdmodel.graphics.state.PDExtendedGraphicsState;

public final class PdfWatermark {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: PdfWatermark input.pdf output.pdf");
      System.exit(1);
    }
    watermark(new File(args[0]), new File(args[1]), "CONFIDENTIAL");
  }

  public static void watermark(File inputPdf, File outputPdf, String text) throws IOException {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      PDExtendedGraphicsState gs = new PDExtendedGraphicsState();
      gs.setNonStrokingAlphaConstant(0.18f);

      for (PDPage page : doc.getPages()) {
        PDRectangle mb = page.getMediaBox();
        float centerX = mb.getWidth() / 2f;
        float centerY = mb.getHeight() / 2f;

        try (PDPageContentStream cs =
            new PDPageContentStream(doc, page, PDPageContentStream.AppendMode.APPEND, true, true)) {
          cs.setGraphicsStateParameters(gs);
          cs.setNonStrokingColor(new Color(10, 10, 10));
          cs.beginText();
          cs.setFont(PDType1Font.HELVETICA_BOLD, 72);
          cs.setTextMatrix(org.apache.pdfbox.util.Matrix.getRotateInstance(Math.toRadians(35), centerX, centerY));
          cs.showText(text);
          cs.endText();
        }
      }
      doc.save(outputPdf);
    }
  }
}

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.font.PDType1Font;

public final class PdfPageNumbers {
  public static void main(String[] args) throws Exception {
    if (args.length < 2) {
      System.out.println("Usage: PdfPageNumbers input.pdf output.pdf");
      System.exit(1);
    }
    addNumbers(new File(args[0]), new File(args[1]));
  }

  public static void addNumbers(File inputPdf, File outputPdf) throws IOException {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      int total = doc.getNumberOfPages();
      int i = 1;
      for (PDPage page : doc.getPages()) {
        String text = "Page " + i + " of " + total;
        float fontSize = 10f;
        float margin = 18f;
        float width = PDType1Font.HELVETICA.getStringWidth(text) / 1000f * fontSize;
        float x = page.getMediaBox().getWidth() - margin - width;
        float y = margin;

        try (PDPageContentStream cs =
            new PDPageContentStream(doc, page, PDPageContentStream.AppendMode.APPEND, true, true)) {
          cs.beginText();
          cs.setFont(PDType1Font.HELVETICA, fontSize);
          cs.newLineAtOffset(x, y);
          cs.showText(text);
          cs.endText();
        }
        i++;
      }
      doc.save(outputPdf);
    }
  }
}

PDF Signer (image placement) (PDFBox)

Build a Java tool that takes an image file (signature.png) and places it at specific X,Y coordinates on the last page of a PDF.

package com.nutrithrive.labs;

import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.graphics.image.PDImageXObject;

public final class PdfImageSigner {
  public static void main(String[] args) throws Exception {
    if (args.length < 6) {
      System.out.println("Usage: PdfImageSigner input.pdf output.pdf signature.png x y width");
      System.exit(1);
    }
    File in = new File(args[0]);
    File out = new File(args[1]);
    File sig = new File(args[2]);
    float x = Float.parseFloat(args[3]);
    float y = Float.parseFloat(args[4]);
    float w = Float.parseFloat(args[5]);
    sign(in, out, sig, x, y, w);
  }

  public static void sign(File inputPdf, File outputPdf, File signaturePng, float x, float y, float width) throws IOException {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      int last = doc.getNumberOfPages() - 1;
      PDPage page = doc.getPage(last);
      PDImageXObject img = PDImageXObject.createFromFileByContent(signaturePng, doc);
      float height = (img.getHeight() * width) / (float) img.getWidth();

      try (PDPageContentStream cs =
          new PDPageContentStream(doc, page, PDPageContentStream.AppendMode.APPEND, true, true)) {
        cs.drawImage(img, x, y, width, height);
      }
      doc.save(outputPdf);
    }
  }
}

Barcode/QR Generator (ZXing + PDFBox)

Develop a Java tool using ZXing and PDFBox that generates a QR code from a URL and embeds it onto the first page of a PDF.

package com.nutrithrive.labs;

import com.google.zxing.BarcodeFormat;
import com.google.zxing.MultiFormatWriter;
import com.google.zxing.common.BitMatrix;
import com.google.zxing.client.j2se.MatrixToImageWriter;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.graphics.image.LosslessFactory;
import org.apache.pdfbox.pdmodel.graphics.image.PDImageXObject;

public final class PdfQrEmbed {
  public static void main(String[] args) throws Exception {
    if (args.length < 6) {
      System.out.println("Usage: PdfQrEmbed input.pdf output.pdf url x y size");
      System.exit(1);
    }
    embed(new File(args[0]), new File(args[1]), args[2], Float.parseFloat(args[3]), Float.parseFloat(args[4]), Integer.parseInt(args[5]));
  }

  public static void embed(File inputPdf, File outputPdf, String url, float x, float y, int sizePx) throws IOException {
    BufferedImage qr = makeQr(url, sizePx);
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      PDPage first = doc.getPage(0);
      PDImageXObject qrImg = LosslessFactory.createFromImage(doc, qr);
      try (PDPageContentStream cs =
          new PDPageContentStream(doc, first, PDPageContentStream.AppendMode.APPEND, true, true)) {
        cs.drawImage(qrImg, x, y, 120, 120);
      }
      doc.save(outputPdf);
    }
  }

  private static BufferedImage makeQr(String text, int size) throws IOException {
    try {
      BitMatrix m = new MultiFormatWriter().encode(text, BarcodeFormat.QR_CODE, size, size);
      return MatrixToImageWriter.toBufferedImage(m);
    } catch (Exception e) {
      throw new IOException("Failed to generate QR", e);
    }
  }
}

Text Redactor (PDFBox)

Write a Java script that searches for a specific regex (like email addresses) in a PDF and draws a black rectangle over those coordinates to redact them.

Note: true redaction needs text-position extraction. Below is a working approach using PDFTextStripper to detect matching text positions and draw rectangles. 

package com.nutrithrive.labs;

import java.awt.Color;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import org.apache.pdfbox.pdmodel.PDDocument;
import org.apache.pdfbox.pdmodel.PDPage;
import org.apache.pdfbox.pdmodel.PDPageContentStream;
import org.apache.pdfbox.pdmodel.PDPageContentStream.AppendMode;
import org.apache.pdfbox.text.PDFTextStripper;
import org.apache.pdfbox.text.TextPosition;

public final class PdfRegexRedactor {
  public static void main(String[] args) throws Exception {
    if (args.length < 3) {
      System.out.println("Usage: PdfRegexRedactor input.pdf output.pdf regex");
      System.exit(1);
    }
    redact(new File(args[0]), new File(args[1]), Pattern.compile(args[2]));
  }

  public static void redact(File inputPdf, File outputPdf, Pattern pattern) throws Exception {
    try (PDDocument doc = PDDocument.load(inputPdf)) {
      for (int pageIndex = 0; pageIndex < doc.getNumberOfPages(); pageIndex++) {
        PDPage page = doc.getPage(pageIndex);

        PositionStripper stripper = new PositionStripper(pattern);
        stripper.setStartPage(pageIndex + 1);
        stripper.setEndPage(pageIndex + 1);
        stripper.getText(doc); // triggers processing

        try (PDPageContentStream cs = new PDPageContentStream(doc, page, AppendMode.APPEND, true, true)) {
          cs.setNonStrokingColor(Color.BLACK);
          for (Rect r : stripper.rects) {
            cs.addRect(r.x, r.y, r.w, r.h);
            cs.fill();
          }
        }
      }
      doc.save(outputPdf);
    }
  }

  private static final class Rect {
    final float x, y, w, h;
    Rect(float x, float y, float w, float h) { this.x = x; this.y = y; this.w = w; this.h = h; }
  }

  private static final class PositionStripper extends PDFTextStripper {
    private final Pattern pattern;
    private final StringBuilder pageText = new StringBuilder();
    private final List<TextPosition> positions = new ArrayList<>();
    final List<Rect> rects = new ArrayList<>();

    PositionStripper(Pattern pattern) throws IOException {
      this.pattern = pattern;
      setSortByPosition(true);
    }

    @Override
    protected void writeString(String string, List<TextPosition> textPositions) throws IOException {
      pageText.append(string);
      positions.addAll(textPositions);
    }

    @Override
    protected void endPage(PDPage page) throws IOException {
      String txt = pageText.toString();
      Matcher m = pattern.matcher(txt);
      while (m.find()) {
        int start = m.start();
        int end = m.end();
        if (start < 0 || end > positions.size() || start >= end) continue;

        // Compute a bounding box across involved glyphs
        float minX = Float.MAX_VALUE, minY = Float.MAX_VALUE;
        float maxX = 0, maxY = 0;
        for (int i = start; i < end; i++) {
          TextPosition tp = positions.get(i);
          float x = tp.getXDirAdj();
          float y = tp.getYDirAdj();
          float w = tp.getWidthDirAdj();
          float h = tp.getHeightDir();
          minX = Math.min(minX, x);
          minY = Math.min(minY, y);
          maxX = Math.max(maxX, x + w);
          maxY = Math.max(maxY, y + h);
        }
        rects.add(new Rect(minX, minY - 1, (maxX - minX), (maxY - minY) + 2));
      }
      pageText.setLength(0);
      positions.clear();
      super.endPage(page);
    }
  }
}
